// 递归搜索算法
bool search(Node* node, int val) {
  if (node == nullptr) return false;
  
  if (node->val == val) {
    return true; // 命中
  }
  
  if (val < node->val) {
    return search(node->left, val);
  } else {
    return search(node->right, val);
  }
}
