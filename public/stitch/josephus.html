void solveJosephus(int n, int k) {
  // 创建循环链表
  Node* head = new Node(1);
  Node* prev = head;
  for (int i = 2; i <= n; i++) {
    prev->next = new Node(i);
    prev = prev->next;
  }
  prev->next = head; // 形成闭环

  // 开始淘汰逻辑
  Node* ptr1 = head, *ptr2 = head;
  while (ptr1->next != ptr1) {
    for (int i = 0; i < k - 1; i++) {
      ptr2 = ptr1;
      ptr1 = ptr1->next;
    }
    ptr2->next = ptr1->next;
    delete ptr1;
    ptr1 = ptr2->next;
  }
}
