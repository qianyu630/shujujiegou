// 汉诺塔递归算法实现
void hanoi(int n, char a, char b, char c) {
if (n == 1) {
move(a, c);
return;
}
// 将 n-1 个盘子从 A 移动到 B
hanoi(n - 1, a, c, b);
// 将最大的盘子从 A 移动到 C
move(a, c);
// 将 n-1 个盘子从 B 移动到 C
hanoi(n - 1, b, a, c);
}
